    <script>
        const dailyVolumeInput = document.getElementById('dailyVolume');
        const errorRateInput = document.getElementById('errorRate');
        const workDaysInput = document.getElementById('workDays');
        const staffCountInput = document.getElementById('staffCount');
        const staffSalaryInput = document.getElementById('staffSalary');

        function updateCalculator() {
            const dailyVolume = parseInt(dailyVolumeInput.value);
            const errorRate = parseFloat(errorRateInput.value);
            const workDays = parseInt(workDaysInput.value) || 0;
            const staffCount = parseInt(staffCountInput.value) || 0;
            const staffSalary = parseInt(staffSalaryInput.value) || 0;

            // 1. 基本單量計算
            const monthlyOrders = dailyVolume * workDays; 
            const monthlyErrors = Math.round((monthlyOrders * errorRate) / 100);
            
            // 2. 成本細項參數 (基於 user 數據估算之單價)
            const unitCosts = {
                shipping: 84,    // 重新配送
                product: 200,    // 商品損耗
                packaging: 20,   // 包材
                repack: 55,      // 重新包裝
                csTime: 75,      // 客服查證
                whTime: 70,      // 倉庫查證
                returns: 105     // 退換貨拒收
            };

            // 3. 計算拆解清單
            const breakdown = {
                '重新配送運費': unitCosts.shipping * monthlyErrors,
                '商品成本 / 損耗': unitCosts.product * monthlyErrors,
                '包材成本': unitCosts.packaging * monthlyErrors,
                '重新包裝人力': unitCosts.repack * monthlyErrors,
                '客服查證時間': unitCosts.csTime * monthlyErrors,
                '倉庫查證時間': unitCosts.whTime * monthlyErrors,
                '退換貨 / 拒收': unitCosts.returns * monthlyErrors
            };

            // 4. UI 更新數據顯示
            document.getElementById('dailyVolumeText').innerText = dailyVolume.toLocaleString() + ' 單';
            document.getElementById('errorRateText').innerText = errorRate + ' 件';

            let breakdownHtml = '';
            for (const [key, val] of Object.entries(breakdown)) {
                breakdownHtml += `
                    <div class="flex justify-between items-center">
                        <span class="text-slate-500">${key} <span class="text-[10px] bg-slate-200 px-1.5 py-0.5 rounded text-slate-600 ml-1">${monthlyErrors} 件</span></span>
                        <span class="font-bold text-slate-700">NT$${val.toLocaleString()}</span>
                    </div>
                `;
            }
            document.getElementById('costBreakdown').innerHTML = breakdownHtml;

            // 5. 總計與 ROI 計算
            const totalLossMid = monthlyErrors * 609; // 基於單次平均隱性成本 609 元
            const dobybotCost = monthlyOrders * 1.5;   // 錄影費 1.5 元
            const monthlySavings = totalLossMid - dobybotCost;
            
            const hourlyWage = Math.round(staffSalary / (workDays * 8));
            const disputeHours = Math.round(monthlyErrors * 1.5); // 假設每單處理需 1.5 小時
            const disputeLaborCost = disputeHours * hourlyWage;

            // 回本期計算 (假設建置開通費 15,000 元)
            const setupFee = 15000;
            const paybackPeriod = monthlySavings > 0 ? (setupFee / monthlySavings).toFixed(1) : '∞';

            // 更新結果卡片
            document.getElementById('totalLossRange').innerText = `NT$${totalLossMid.toLocaleString()}`;
            document.getElementById('costBefore').innerText = `NT$${totalLossMid.toLocaleString()}`;
            document.getElementById('costAfter').innerText = `NT$${dobybotCost.toLocaleString()}`;
            document.getElementById('monthlySavingsText').innerText = `NT$${Math.max(0, monthlySavings).toLocaleString()}`;
            document.getElementById('paybackPeriod').innerText = (monthlySavings > 0 ? (paybackPeriod > 12 ? '12+' : paybackPeriod) : '∞') + ' 個月';
            
            document.getElementById('disputeHours').innerText = `約 ${disputeHours} 小時`;
            document.getElementById('disputeLaborCost').innerText = `NT$${disputeLaborCost.toLocaleString()}`;

            const staffSaved = (monthlySavings / staffSalary).toFixed(1);
            document.getElementById('summaryConclusion').innerHTML = 
                `以您目前的規模，每月問題訂單導致的隱性損失約 <b>NT$${totalLossMid.toLocaleString()}</b>，而 Dobybot 的錄影費用僅需 <b>NT$${dobybotCost.toLocaleString()}</b>。導入後每月可節省約 <b>NT$${Math.max(0, monthlySavings).toLocaleString()}</b>，這相當於省下了 <b>${staffSaved} 位</b> 包貨人員的人事預算。系統不只是工具，更是您利潤的守護者。`;
        }

        // 監聽所有輸入變動
        [dailyVolumeInput, errorRateInput, workDaysInput, staffCountInput, staffSalaryInput].forEach(input => {
            input.addEventListener('input', updateCalculator);
        });

        // 頁面加載時執行一次
        updateCalculator();
    </script>
